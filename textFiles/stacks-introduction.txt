\begin{lemma}[Key fact] \label{lemma-key-fact} The functor $\Sch^{opp} \to \textit{Sets}$, $T \mapsto \{(a, a', \alpha)\text{ as above}\}$ is representable by a scheme $S \times_{\mathcal{M}_{1, 1}} S'$. \end{lemma} 
\begin{definition} \label{definition-smooth} We say a morphism $S \to \mathcal{M}_{1, 1}$ is {\it smooth} if for every morphism $S' \to \mathcal{M}_{1, 1}$ the projection morphism $$ S \times_{\mathcal{M}_{1, 1}} S' \longrightarrow S' $$ is smooth. \end{definition} 
\begin{definition} \label{definition-algebraic-stack} We say $\mathcal{M}_{1, 1}$ is an {\it algebraic stack} if and only if \begin{enumerate} \item We have descent for objects for the \'etale topology on $\Sch$. \item The key fact holds. \item there exists a surjective and smooth morphism $S \to \mathcal{M}_{1, 1}$. \end{enumerate} \end{definition} 
\begin{lemma} \label{lemma-Weierstrass-smooth-cover} The morphism $W \xrightarrow{(E_W, f_W, 0_W)} \mathcal{M}_{1, 1}$ is smooth and surjective. \end{lemma}
